it add <file> - добавления файла в локальный репозиторий
git add <folder/> - добавления рекурсивно папки в локальный репозиторий
git add . - добавления все в локальный репозиторий
git status - статус файлов
git commit -m - комит файлов
git log - просмотр изменений
git log - удобный просмотр изменений
.gitignore - файл для добалвения исключений
git checkout <ver> - просмотр текущей версии файла
git revert <ver> -  отмена комита
git branch <name> - создание ветки
git branch -a - список веток
git checkout <name> - переключение на ветку
git merge <name> - обьединение веток
